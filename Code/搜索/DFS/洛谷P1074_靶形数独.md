#include <iostream>
#include <vector>

const int N = 15;
int a[N][N];
int res = -1;

inline void DFS ( int i, int j ) {
        if ( j >= 9 ) {
                int cur = 0;
                for ( int x = 0; x < 9; x ++ ) {
                        for ( int y = 0; y < 9; y ++ ) {
                                if      ( x >= 4 && x <= 4 && y >= 4 && y <= 4 ) cur += a[x][y] * 10;
                                else if ( x >= 3 && x <= 5 && y >= 3 && y <= 5 ) cur += a[x][y] * 9;
                                else if ( x >= 2 && x <= 6 && y >= 2 && y <= 6 ) cur += a[x][y] * 8;
                                else if ( x >= 1 && x <= 7 && y >= 1 && y <= 7 ) cur += a[x][y] * 7;
                                else                                             cur += a[x][y] * 6;
                        }
                }

                res = max ( res, cur );
                return;
        }
