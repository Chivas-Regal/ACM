```cpp
const int N = 1e5 + 10;
const int mxBt = 64;
ll n, m;
ll d[mxBt];

inline int Insert ( ll x ) {
        for ( int j = mxBt - 1; j >= 0; j -- ) {
                if ( x & (1ll << j) ) {
                        if ( d[j] == 0 ) { d[j] = x; return j; }
                        else               x ^= d[j];
                }
        }
        return -1;
}

int main () {
        ios::sync_with_stdio(false);
        cin >> n; 
        for ( ll i = 1, x; i <= n; i ++ )
                cin >> x,
                Insert ( x );
        cin >> m;
        for ( ll i = 1; i <= m; i ++ ) {
                ll x, y; cin >> x >> y;
                int chg = Insert ( x ^ y );
                if ( chg == -1 )          cout << "YES" << endl;
                else                      cout << "NO" << endl, d[chg] = 0;
        }
}
```
