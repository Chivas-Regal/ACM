const int N =  1e5 + 10;
int a[N], n;
ll F[N];

namespace Number {
        int mu[N];
        bool not_prime[N];
        vector<int> prime;
        inline void Sieve () {
                not_prime[0] = not_prime[1] = mu[1] = 1;
                for ( int i = 2; i < N; i ++ ) {
                        if ( !not_prime[i] ) 
                                prime.push_back(i),
                                mu[i] = -1;
                        for ( int j = 0; j < prime.size() && i * prime[j] < N; j ++ ) {
                                not_prime[i * prime[j]] = true;
                                if ( i % prime[j] == 0 ) break; 
                                mu[i * prime[j]] = -mu[i];
                        }
                }
        }

        ll mark[N];
        inline void Pre () {
                for ( int i = 0; i < n; i ++ ) mark[a[i]] += a[i];
                for ( int d = 1; d < N; d ++ ) {
                        for ( int i = d; i < N; i += d ) {
                                F[d] += mark[i];
                        }
                        F[d] *= F[d];
                }
        }
} using namespace Number;

int main () {
        cin >> n; for ( int i = 0; i < n; i ++ ) cin >> a[i];
        Pre (); Sieve ();
        ll res = 0;
        for ( int k = 1; k < N; k ++ ) {
                ll cur = 0;
                for ( int d = 1; d < N / k; d ++ ) cur += mu[d] * F[d * k];
                res += cur / k;
        }
        cout << res << endl;
}
