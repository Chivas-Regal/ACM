using namespace std;
const int maxn = 1e6+10;
struct Trie{
    int nxt;
    int cnt;
}trie[maxn][26];
int tot = 1;

void insert(string s){
    int deep = 1;
    for (int i = 0; i < s.size(); i++){
        int cur_c = s[i] - 'a';
        if(!trie[deep][cur_c].nxt)
            trie[deep][cur_c].nxt = ++tot;
        deep = trie[deep][cur_c].nxt;
        if (i == s.size() - 1)
            trie[deep][cur_c].cnt++;
    }
}
int search(string s){
    int deep = 1;
    int num=0;
    for (int i = 0; i < s.size(); i++){
        deep = trie[deep][s[i] - 'a'].nxt;
        if(deep==0)
            break;
        num += trie[deep][s[i] - 'a'].cnt;
    }
    return num;
}

int main()
{
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < n; i++){
        string s;
        cin >> s;
        insert(s);
    }
    for (int i = 0; i < m; i++){
        string s;
        cin >> s;
        cout<<search(s)<<endl;
    }
    return 0;
}
