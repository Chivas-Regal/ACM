const int N = 1e6 + 10;

class KMP_node{
private:
        string a, b;
        int nxt[N];

public:
        inline KMP_node ( string aa, string bb ) {
                a = aa;
                b = bb;
        }
        inline void Get_Next ( ) {
                for ( int i = 0; i < N; i ++ ) nxt[i] = -1;
                int j = -1, m = b.size();
                for ( int i = 0; i + 1 < m; i ++ ) {
                        while ( j >= 0 && b[i + 1] != b[j + 1] ) j = nxt[j];
                        if ( b[i + 1] == b[j + 1] ) j ++;
                        nxt[i + 1] = j;
                }
        }
        inline int KMP ( ) {
                int n = a.size(), m = b.size();
                int j = -1, res = 0;
                for ( int i = -1; i + 1 < n; i ++ ) {
                        while ( j >= 0 && a[i + 1] != b[j + 1] ) j = nxt[j];
                        if ( a[i + 1] == b[j + 1]) j ++;
                        if ( j == m - 1 ) {
                                res ++;
                                j = nxt[j];
                        }
                }
                return res;
        }
};

string sa, sb;
